// ════════════════════════════════════════════════════ // TIMERPRO - Google Apps Script v7 (POST + CORS) // ════════════════════════════════════════════════════ const FILE_NAME = 'TimerPro_Datos.json'; const TEMP_FILE = 'TimerPro_Temp.json'; function doGet(e) { try { const callback = e.parameter.callback || ''; const action = e.parameter.action || 'load'; let result; if (action === 'ping') { result = { ok: true, pong: true, ts: new Date().toISOString() }; } else if (action === 'save') { const raw = e.parameter.d; if (!raw) { result = { ok: false, error: 'No data' }; } else { const data = JSON.parse(raw); saveData(data); result = { ok: true, saved: true, entries: (data.entries||[]).length }; } } else if (action === 'save_meta') { const raw = e.parameter.d; if (!raw) { result = { ok: false, error: 'No meta data' }; } else { const meta = JSON.parse(raw); meta.entries = []; saveTempData(meta); result = { ok: true, meta_saved: true }; } } else if (action === 'save_entry') { const idx = parseInt(e.parameter.idx || '0'); const raw = e.parameter.d; if (!raw) { result = { ok: false, error: 'No entry data' }; } else { const entry = JSON.parse(raw); const temp = loadTempData(); if (!temp.entries) temp.entries = []; temp.entries[idx] = entry; saveTempData(temp); result = { ok: true, entry_saved: idx }; } } else if (action === 'commit') { const temp = loadTempData(); temp.lastSync = new Date().toISOString(); saveData(temp); result = { ok: true, committed: true, entries: (temp.entries||[]).length }; } else if (action === 'load') { const data = loadData(); result = { ok: true, data: data }; } else { result = { ok: true, action: action }; } return buildResponse(JSON.stringify(result), callback); } catch(err) { const cb = (e.parameter && e.parameter.callback) || ''; return buildResponse(JSON.stringify({ ok: false, error: err.message }), cb); } } // POST handler with CORS - accepts large payloads function doPost(e) { try { const raw = (e.postData && e.postData.contents) || (e.parameter && (e.parameter.d || e.parameter.data)); if (!raw) throw new Error('No data'); const data = JSON.parse(raw); saveData(data); const result = JSON.stringify({ ok: true, saved: true, entries: (data.entries||[]).length }); return ContentService.createTextOutput(result) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput( JSON.stringify({ ok: false, error: err.message }) ).setMimeType(ContentService.MimeType.JSON); } } function buildResponse(json, callback) { if (callback) { return ContentService.createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } function getOrCreateFile(name) { const files = DriveApp.getFilesByName(name); if (files.hasNext()) return files.next(); return DriveApp.createFile(name, '{}', MimeType.PLAIN_TEXT); } function loadData() { try { return JSON.parse(getOrCreateFile(FILE_NAME).getBlob().getDataAsString()) || {}; } catch(e) { return {}; } } function saveData(data) { getOrCreateFile(FILE_NAME).setContent(JSON.stringify(data)); } function loadTempData() { try { return JSON.parse(getOrCreateFile(TEMP_FILE).getBlob().getDataAsString()) || {}; } catch(e) { return {}; } } function saveTempData(data) { getOrCreateFile(TEMP_FILE).setContent(JSON.stringify(data)); }
